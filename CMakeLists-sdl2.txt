cmake_minimum_required(VERSION 3.10)
project(OpenMugen)

add_definitions("-Wall")
file(GLOB_RECURSE SOURCE_FILES "src/*.cpp")

set_property(TARGET ${PROJECT} PROPERTY CXX_STANDARD 99)
set_property(TARGET ${PROJECT} PROPERTY CXX_STANDARD_REQUIRED 99)

INCLUDE(FindPkgConfig REQUIRED)
PKG_CHECK_MODULES(SDL2 REQUIRED sdl2)
PKG_CHECK_MODULES(SDL2_IMAGE REQUIRED SDL2_image)
PKG_CHECK_MODULES(SDL2_GFX REQUIRED SDL2_gfx)
PKG_CHECK_MODULES(SDL12COMPAT REQUIRED sdl12_compat)

if (CMAKE_BUILD_TYPE MATCHES Debug)
    add_definitions("-g -O0 -Wall -Wextra -Wpedantic -fsanitize=address")
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -fsanitize=address")
endif ()

message("SDL2_INCLUDE_DIRS: ${SDL2_INCLUDE_DIRS}")
message("SDL2_IMAGE_INCLUDE_DIRS: ${SDL2_IMAGE_INCLUDE_DIRS}")
message("SDL2_GFX_INCLUDE_DIRS: ${SDL2_GFX_INCLUDE_DIRS}")
message("SDL12COMPAT_INCLUDE_DIRS: ${SDL12COMPAT_INCLUDE_DIRS}")
include_directories(${SDL2_INCLUDE_DIRS} ${SDL2_IMAGE_INCLUDE_DIRS} ${SDL2_GFX_INCLUDE_DIRS} ${SDL12COMPAT_INCLUDE_DIRS})

message("SDL2_LIBRARIES: ${SDL2_LIBRARIES}")
message("SDL2_IMAGE_LIBRARIES: ${SDL2_IMAGE_LIBRARIES}")
message("SDL2_GFX_LINK_LIBRARIES: ${SDL2_GFX_LINK_LIBRARIES}")
message("SDL12COMPAT_LIBRARIES: ${SDL12COMPAT_LIBRARIES}")
message("SDL2_LINK_LIBRARIES: ${SDL2_LINK_LIBRARIES}")
message("SDL2_IMAGE_LINK_LIBRARIES: ${SDL2_IMAGE_LINK_LIBRARIES}")
target_link_libraries(${PROJECT} ${SDL2_LIBRARIES} ${SDL2_IMAGE_LIBRARIES} ${SDL2_GFX_LINK_LIBRARIES} ${SDL12COMPAT_LIBRARIES})
